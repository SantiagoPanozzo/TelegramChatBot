175 public string ParseCedula(string cedula)
176 {
177 string validos = "0123456789";
178 string parsed = String.Empty;
179
180 foreach (char caracteres in cedula)
181 {
182 if (validos.Contains(caracteres)) parsed += caracteres;
183 }
184
185 switch (parsed.Length)
186 {
187 case (7):
188 parsed = parsed.Insert(3, ".");
189 parsed = parsed.Insert(7, "-");
190 break;
191 case (8):
192 parsed = parsed.Insert(1, ".");
193 parsed = parsed.Insert(5, ".");
194 parsed = parsed.Insert(9, "-");
195 break;
196 default:
197 break;
198 }
199
200 return parsed;
201 }

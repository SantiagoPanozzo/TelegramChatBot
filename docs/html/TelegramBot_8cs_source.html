55private static void Start()
56{
57 // Lee una variable de entorno NETCORE_ENVIRONMENT que si no existe o tiene el valor 'development' indica
58 // que estamos en un ambiente de desarrollo.
59 var developmentEnvironment = Environment.GetEnvironmentVariable("NETCORE_ENVIRONMENT");
60 var isDevelopment =
61 string.IsNullOrEmpty(developmentEnvironment) ||
62 developmentEnvironment.ToLower() == "development";
63
64 var builder = new ConfigurationBuilder();
65 builder
66 .SetBasePath(AppDomain.CurrentDomain.BaseDirectory);
67 //.AddJsonFile("appsettings.json", optional: false, reloadOnChange: true);
68
69 // En el ambiente de desarrollo el token secreto del bot se toma de la configuración secreta
70 if (isDevelopment)
71 {
72 builder.AddUserSecrets<TelegramBot>();
73 }
74
75 var configuration = builder.Build();
76
77 IServiceCollection services = new ServiceCollection();
78
79 // Mapeamos la implementación de las clases para inyección de dependencias
80 services
81 .Configure<BotSecret>(configuration.GetSection(nameof(BotSecret)))
82 .AddSingleton<ISecretService, SecretService>();
83
84 var serviceProvider = services.BuildServiceProvider();
85 var revealer = serviceProvider.GetService<ISecretService>();
86 token = revealer.Token;
87}

130public static async Task HandleUpdateAsync(ITelegramBotClient botClient, Update update, CancellationToken cancellationToken)
131{
132 try
133 {
134 // Sólo respondemos a mensajes de texto
135 if (update.Type == UpdateType.Message)
136 {
137 await HandleMessageReceived(botClient, update.Message);
138 }
139 }
140 catch(Exception e)
141 {
142 await HandleErrorAsync(botClient, e, cancellationToken);
143 }
144}

149private static async Task HandleMessageReceived(ITelegramBotClient botClient, Message message)

151 // Estas tres líneas es para serializar message a ver que trae.

153 string jsonString = JsonSerializer.Serialize(message, options);
154 // Console.WriteLine(jsonString);

156 Console.WriteLine($"Se recibió un mensaje de {message.From.FirstName} consultando por: {message.Text}");
157
158 string response = string.Empty;
159
160 firstHandler.Handle(message, out response);
161
162 if (!string.IsNullOrEmpty(response))
163 {
164 await Bot.SendTextMessageAsync(message.Chat.Id, response);
165 }
166}
